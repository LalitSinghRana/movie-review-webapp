# movie-review-site

# set up DB
  - crate DB "movie-review-site" in your localhost DB before running server
  - if not using MySQL, 
    - change "dialect" in server/src/database/config.ts to whatever SQL DB you're using

# run server
  - "npm run dev"
  - or "tsc --build"
    - then "npm start"

# run client 
  - "npm start"
